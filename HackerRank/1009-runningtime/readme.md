    using System.CodeDom.Compiler;

    class Result
    {

    

        public static void staircase(int n)
        {
                for(int i=0;i<n;i++)
                {
                    for(int j=0;j<n;j++)
                    {
                        Console.Write(" ");
                    }

                    Console.Write("#");
                }
        }

    }

    class Solution
    {
        public static void Main(string[] args)
        {
            int n = Convert.ToInt32(Console.ReadLine().Trim());

            Result.staircase(n);
        }
    }
